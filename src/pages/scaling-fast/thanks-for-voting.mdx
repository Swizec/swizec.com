import {
  ArticleListing,
} from "@swizec/gatsby-theme-course-platform"
import { graphql } from "gatsby"

export const pageQuery = graphql`
  query {
    allMdx(
      filter: {
        internal: { contentFilePath: { regex: "/blog/.+/" } }
        frontmatter: { categories: { regex: "/Scaling Fast Book/gi" } }
      }
      sort: { frontmatter: { published: DESC } }
    ) {
      nodes {
        fields {
          slug
        }
        frontmatter {
          title
          description
          published
        }
      }
    }
  }

`
