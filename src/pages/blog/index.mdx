import { graphql } from "gatsby"
import { ArticleListing } from "../../components/ArticleListing"

export const pageQuery = graphql`
  query {
    allSitePage(
      filter: { path: { regex: "/blog/.+/" } }
      sort: { fields: context___frontmatter___published, order: DESC }
    ) {
      nodes {
        path
        context {
          frontmatter {
            title
            description
            published
          }
        }
      }
    }
  }
`
